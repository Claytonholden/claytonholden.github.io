Project 1 – CLI Calculator

A simple calculator that runs in the terminal. The user picks an operation and enters two numbers. The script validates input, performs the math, and loops until the user chooses to quit.

File: PythonEssentials/projects/cli-calculator.py

View calculator code 
# cli-calculator.py

def get_number(prompt: str) -> float:
    """Prompt until the user enters a valid number."""
    while True:
        raw = input(prompt).strip()
        try:
            return float(raw)
        except ValueError:
            print(f"  [!] '{raw}' is not a valid number. Try again.")

def get_operation() -> str:
    """Return one of + - * / or q to quit."""
    valid = {"+", "-", "*", "/", "q"}
    while True:
        op = input("Choose operation (+, -, *, /) or q to quit: ").strip()
        if op in valid:
            return op
        print("  [!] Invalid choice. Use + - * / or q.")

def calculate(a: float, b: float, op: str) -> float:
    if op == "+": return a + b
    if op == "-": return a - b
    if op == "*": return a * b
    if op == "/":
        if b == 0:
            raise ZeroDivisionError("Cannot divide by zero.")
        return a / b
    raise ValueError(f"Unsupported operation {op!r}")

def main() -> None:
    print("=== CLI Calculator ===")
    while True:
        op = get_operation()
        if op == "q":
            print("Goodbye.")
            break

        a = get_number("First number:  ")
        b = get_number("Second number: ")

        try:
            result = calculate(a, b, op)
            print(f"Result: {a} {op} {b} = {result}\n")
        except ZeroDivisionError as exc:
            print(f"  [!] Error: {exc}\n")

if __name__ == "__main__":
    main()

Project 2 – Decision Helper

A branching decision helper that asks a series of questions and prints a recommendation. It can be adapted for troubleshooting, study planning, or simple “what should I do next?” flows.

File: PythonEssentials/projects/decision-helper.py

View decision helper code 
# decision-helper.py

def ask_yes_no(prompt: str) -> bool:
    while True:
        ans = input(prompt + " [y/n]: ").strip().lower()
        if ans in {"y", "yes"}:
            return True
        if ans in {"n", "no"}:
            return False
        print("  [!] Please answer with y or n.")

def main() -> None:
    print("=== Decision Helper ===")
    print("This mini expert system walks through a few questions and suggests a next step.\n")

    tired = ask_yes_no("Are you mentally tired?")
    blocked = ask_yes_no("Are you stuck on a technical problem?")
    deadline_soon = ask_yes_no("Do you have a deadline within 24 hours?")

    if deadline_soon and blocked:
        recommendation = "Take a 5-minute break, then search docs / error messages and ask for help if still stuck."
    elif deadline_soon and not blocked:
        recommendation = "Focus on the deliverable. Avoid distractions until the main task is done."
    elif not deadline_soon and tired:
        recommendation = "Step away, grab water or food, and come back later with a fresh mind."
    else:
        recommendation = "Use this time to practice or refactor an existing project."

    print("\nSuggested next step:")
    print(f"- {recommendation}")

if __name__ == "__main__":
    main()

Project 3 – Loop & List Lab

This lab focuses on loops and lists: the script collects items from the user, stores them in a list, and prints a small summary report.

File: PythonEssentials/projects/loop-list-lab.py

View loop & list lab code 
# loop-list-lab.py

def collect_items() -> list[str]:
    print("Enter tools you use regularly (blank line to finish):")
    items: list[str] = []
    while True:
        entry = input("Tool: ").strip()
        if not entry:
            break
        items.append(entry)
    return items

def show_summary(items: list[str]) -> None:
    print("\n=== Tool Summary ===")
    if not items:
        print("No tools entered.")
        return

    print(f"Total tools: {len(items)}")
    print("Unique tools:", len(set(item.lower() for item in items)))
    print("\nList:")
    for i, item in enumerate(items, start=1):
        print(f"{i:2}. {item}")

def main() -> None:
    tools = collect_items()
    show_summary(tools)

if __name__ == "__main__":
    main()

Project 4 – Functions Lab

This lab emphasizes breaking logic into small functions and using a clear main() entry point. The example here is a tiny grade calculator.

File: PythonEssentials/projects/functions-lab.py

View functions lab code 
# functions-lab.py

from typing import Sequence

def average(values: Sequence[float]) -> float:
    if not values:
        raise ValueError("Cannot average an empty list.")
    return sum(values) / len(values)

def get_scores() -> list[float]:
    scores: list[float] = []
    print("Enter assignment scores (0–100). Blank line to finish.")
    while True:
        raw = input("Score: ").strip()
        if not raw:
            break
        try:
            value = float(raw)
            if 0 <= value <= 100:
                scores.append(value)
            else:
                print("  [!] Score must be between 0 and 100.")
        except ValueError:
            print("  [!] Please enter a number.")
    return scores

def letter_grade(score: float) -> str:
    if score >= 90: return "A"
    if score >= 80: return "B"
    if score >= 70: return "C"
    if score >= 60: return "D"
    return "F"

def main() -> None:
    print("=== Grade Calculator ===")
    scores = get_scores()
    if not scores:
        print("No scores entered.")
        return

    avg = average(scores)
    print(f"\nAverage: {avg:.2f}%  (Letter grade: {letter_grade(avg)})")

if __name__ == "__main__":
    main()

Service Learning Project – Baseball Walkup Song Player

The capstone project for this course is a command-line Baseball Walkup Song Player. It is designed to run on a laptop at the field and quickly trigger the correct walkup song for each player.

Project folder: /projects/WalkupSongPlayer/ (main script and supporting files).

Features

  • Stores a roster of players and their walkup songs in a simple data structure.
  • Displays a numbered player menu with search by jersey number or name.
  • Plays (or simulates playing) the correct audio track when a player is selected.
  • Logs each play so you can review usage after the game.
View example walkup player code 
# walkup_song_player.py (simplified example)

from dataclasses import dataclass
from typing import Dict, List
import datetime
import pathlib
import subprocess
import sys

BASE_DIR = pathlib.Path(__file__).resolve().parent
LOG_FILE = BASE_DIR / "walkup_log.txt"

@dataclass
class Player:
    number: int
    name: str
    song_title: str
    file_path: pathlib.Path

ROSTER: Dict[int, Player] = {
    2: Player(2,  "Clayton Holden", "Walkup Anthem",
              BASE_DIR / "audio" / "clayton_walkup.mp3"),
    7: Player(7,  "Player Seven",   "Power Track",
              BASE_DIR / "audio" / "player7.mp3"),
    12: Player(12, "Player Twelve", "Hype Song",
               BASE_DIR / "audio" / "player12.mp3"),
    # add more players here
}

def clear_screen() -> None:
    if sys.platform.startswith("win"):
        subprocess.call("cls", shell=True)
    else:
        subprocess.call("clear", shell=True)

def print_menu() -> None:
    print("=== Baseball Walkup Song Player ===\n")
    print("Select a player by number, or type Q to quit.\n")
    for number, player in sorted(ROSTER.items()):
        print(f"{number:2}  {player.name:20}  -  {player.song_title}")
    print()

def play_song(player: Player) -> None:
    timestamp = datetime.datetime.now().isoformat(timespec="seconds")
    log_line = f"{timestamp}  #{player.number} {player.name}  -  {player.song_title}\n"
    LOG_FILE.write_text(LOG_FILE.read_text() + log_line if LOG_FILE.exists() else log_line)

    print(f"\n[Playing] {player.name} – {player.song_title}")
    print(f"File: {player.file_path}")

    # If you have a CLI audio player installed, uncomment one of these:
    # subprocess.Popen([\"start\", str(player.file_path)], shell=True)  # Windows
    # subprocess.Popen([\"afplay\", str(player.file_path)])             # macOS
    # subprocess.Popen([\"mpv\", str(player.file_path)])                # cross-platform

def main() -> None:
    while True:
        clear_screen()
        print_menu()
        choice = input("Player number (or Q): ").strip().lower()
        if choice in {"q", "quit", "exit"}:
            print("Good game. Exiting.")
            break
        try:
            number = int(choice)
        except ValueError:
            input("\n[!] Invalid choice. Press Enter to continue...")
            continue

        player = ROSTER.get(number)
        if not player:
            input("\n[!] No player with that number. Press Enter to continue...")
            continue

        play_song(player)
        input("\nPress Enter when the batter is finished...")

if __name__ == "__main__":
    main()
